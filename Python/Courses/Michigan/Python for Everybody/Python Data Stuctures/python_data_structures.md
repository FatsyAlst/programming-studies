```
# Exemplo básico de uso do método .get() em dicionários Python

carros = {"marca": "Ford", "modelo": "Mustang"}
print(carros.get("modelo"))      # Saída: Mustang

# Se a chave não existir, retorna None
print(carros.get("preco"))       # Saída: None

# Definindo um valor padrão para retorno caso a chave não exista
print(carros.get("preco", 0))    # Saída: 0

# Usando .get() para contagem de elementos (evitando KeyError)
contagem = {}
lista = ["a", "b", "a", "c", "b", "a"]
for item in lista:
    contagem[item] = contagem.get(item, 0) + 1

print(contagem)  # Saída: {'a': 3, 'b': 2, 'c': 1}

```
